\begin{align}
    V_C&=-V_L\label{eq:423}\\

\frac{10^6}{j\omega_0} &= -10^{-2}j\omega_0 \label{eq:425}\\
\frac{10^6}{10^{-2}} &= -j^2\omega_0^{2} \label{eq:426}\\ 
\omega_0^{2}&=10^8 \label{eq:427}\\
